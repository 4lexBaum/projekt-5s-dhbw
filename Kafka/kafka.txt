Use Kafka on Windows -> 
Bein Entpacken --force-local angeben!

Schritt 1: Kafka downloaden:
=============================
- http://kafka.apache.org/downloads.html
- Bei Ausführung auf Windows: http://stackoverflow.com/questions/23115013/is-there-an-easy-way-to-install-kafka-on-windows
  (Installation von Cygwin64 und anpassen der Datei kafka-run-class.sh -> letzte Zeile)

(folgende Befehle im Kafka Source-Ordner ausführen)
Schritt 2: Serverinstanzen starten
===================================
Zookeeper >> bin/zookeeper-server-start.sh config/zookeeper.properties
Kafka     >> bin/kafka-server-start.sh config/server.properties

Schritt 3: Nachrichten senden
==============================
Built-in program >> bin/kafka-console-producer.sh --zookeeper localhost:2181 --topic test
oder Java verwenden:

package test;

import java.util.Properties;

import kafka.javaapi.producer.Producer;
import kafka.producer.KeyedMessage;
import kafka.producer.ProducerConfig;

@SuppressWarnings("all")
public class ProducerTest {
    public static void main(String[] args) {
        Properties props = new Properties();
        props.put("zk.connect", "localhost:2181");
        props.put("serializer.class", "kafka.serializer.StringEncoder");
        props.put("metadata.broker.list", "localhost:9092");
 
        ProducerConfig config = new ProducerConfig(props);
 
        Producer<String, String> producer = new Producer<String, String>(config);
 
        KeyedMessage<String, String> data = new KeyedMessage<String, String>("demo", "Hallo Welt!");
        producer.send(data);
        producer.close();
    }
}

Schritt 4: Nachrichten konsumieren
==================================
Built-in program >> bin/kafka-console-consumer.sh --zookeeper localhost:2181 --topic test --from-beginning
oder Java verwenden: (Quelle: http://wpcertification.blogspot.de/2014/08/java-client-for-publishing-and.html)

package test;

import java.util.*;

import kafka.consumer.Consumer;
import kafka.consumer.ConsumerConfig;
import kafka.consumer.ConsumerIterator;
import kafka.consumer.KafkaStream;
import kafka.javaapi.consumer.ConsumerConnector;

public class ConsumerTest {
	static ConsumerConnector consumerConnector;
	
	public static void main(String[] args) {
	Properties props = new Properties();
        props.put("zookeeper.connect", "localhost:2181");
        props.put("group.id", "test-group");
        ConsumerConfig consumerConfig = new ConsumerConfig(props);
        consumerConnector = Consumer.createJavaConsumerConnector(consumerConfig);
        
        Map<String, Integer> topicCountMap = new HashMap<String, Integer>();
        topicCountMap.put("demo", new Integer(1));
        Map<String, List<KafkaStream<byte[], byte[]>>> consumerMap = consumerConnector.createMessageStreams(topicCountMap);
        KafkaStream<byte[], byte[]> stream =  consumerMap.get("demo").get(0);
        ConsumerIterator<byte[], byte[]> it = stream.iterator();
        
        while(it.hasNext()) {
            System.out.println(new String(it.next().message()));
        }
    }
}
